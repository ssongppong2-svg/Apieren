-- 1. 프로필 create table profiles ( id uuid references auth.users primary key, nickname text not null, email text, created_at timestamptz default now() ); -- 2. 게시글 (게임/잡담/짤 카테고리) create table posts ( id bigint generated always as identity primary key, user_id uuid references auth.users, nickname text, title text not null, content text, category text default '잡담' check (category in ('게임','잡담','짤')), images text[] default '{}', likes int default 0, dislikes int default 0, views int default 0, created_at timestamptz default now() ); -- 3. 댓글 (대댓글: parent_id 자기참조) create table comments ( id bigint generated always as identity primary key, post_id bigint references posts on delete cascade, user_id uuid references auth.users, parent_id bigint references comments, nickname text, content text not null, created_at timestamptz default now() ); -- 4. 반응 (좋아요/싫어요 — 사용자당 1개 제한) create table reactions ( id bigint generated always as identity primary key, post_id bigint references posts on delete cascade, user_id uuid references auth.users not null, type text check (type in ('like','dislike')), unique(post_id, user_id) ); -- 5. 알림 create table notifications ( id bigint generated always as identity primary key, user_id uuid references auth.users, type text, message text, post_id bigint, is_read boolean default false, created_at timestamptz default now() ); -- 6.
